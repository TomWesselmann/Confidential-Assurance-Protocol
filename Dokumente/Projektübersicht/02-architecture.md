## Datenmodelle

### Core Data Structures

#### Input Models

```rust
// Supplier record (from CSV)
struct Supplier {
    name: String,           // Company name
    jurisdiction: String,   // Country code (e.g., "DE")
    tier: String,          // "TIER_1", "TIER_2", etc.
}

// UBO record (from CSV)
struct Ubo {
    name: String,          // Full name
    birthdate: String,     // RFC3339 format
    citizenship: String,   // Country code
}
```

#### Commitment Models

```rust
// Merkle root commitments
struct Commitments {
    supplier_root: String,              // 0x + 64 hex (BLAKE3)
    ubo_root: String,
    company_commitment_root: String,
    supplier_count: Option<usize>,
    ubo_count: Option<usize>,
}
```

#### Policy Models

```rust
// Compliance policy
struct Policy {
    version: String,                    // "lksg.v1"
    name: String,
    created_at: String,                 // RFC3339
    constraints: PolicyConstraints,
    notes: String,
}

struct PolicyConstraints {
    require_at_least_one_ubo: bool,
    supplier_count_max: u32,
    ubo_count_min: Option<u32>,
    require_statement_roots: Option<Vec<String>>,
}
```

#### Manifest Model

```rust
// Complete proof manifest
struct Manifest {
    version: String,                    // "manifest.v1.0"
    created_at: String,                 // RFC3339
    supplier_root: String,
    ubo_root: String,
    company_commitment_root: String,
    policy: PolicyInfo,
    audit: AuditInfo,
    proof: ProofInfo,
    signatures: Vec<SignatureInfo>,
    time_anchor: Option<TimeAnchor>,    // v0.9.0+
}
```

#### Proof Model

```rust
// Cryptographic proof
struct Proof {
    version: String,                    // "proof.v0"
    type: String,                       // "mock", "zkvm", "halo2"
    statement: String,                  // Policy reference
    manifest_hash: String,              // SHA3-256
    policy_hash: String,
    proof_data: ProofData,
    status: String,                     // "ok", "fail"
}

struct ProofData {
    checked_constraints: Vec<ConstraintCheck>,
}

struct ConstraintCheck {
    name: String,
    ok: bool,
}
```

// Trait fÃ¼r alle Proof-Backends
trait ProofSystem {
    fn backend_name(&self) -> &str;
    fn verify(&self, proof_data: &ProofData, statement: &str) -> Result<bool>;
}

// Implementierungen
struct MockZK;      // Phase 1 (implemented)
struct Halo2ZK;     // Phase 3 (planned)
struct SpartanZK;   // Phase 4 (planned)
struct RiscZeroZK;  // Phase 4 (planned)

// Factory
fn backend_factory(backend: &str) -> Box<dyn ProofSystem> {
    match backend {
        "mock" => Box::new(MockZK),
        "halo2" => Box::new(Halo2ZK),
        _ => panic!("Unknown backend"),
    }
}
